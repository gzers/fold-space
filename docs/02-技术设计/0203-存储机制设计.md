# 褶宇宙 Fold-Space 存储机制文档

**架构原则**：纯前端、无服务器、本地优先、用户完全拥有数据

---

## 一、双轨存储架构

| 层级 | 技术方案 | 用途 | 容量 | 持久性 |
|------|----------|------|------|--------|
| **运行层** | IndexedDB (Dexie.js) | 实时读写、快速查询、状态管理 | 理论上无上限（通常50MB+） | 浏览器级，清理缓存可能丢失 |
| **文件层** | File System Access API | 本地文件持久化、备份、跨设备迁移 | 受限于硬盘空间 | 用户文件系统，永久保存 |
| **降级层** | Blob Download | 兼容性导出/导入（旧浏览器） | 同文件层 | 手动下载管理 |

---

## 二、技术选型

### 2.1 IndexedDB 封装：Dexie.js
```bash
npm install dexie
```

数据库设计：

```javascript
// db.js
import Dexie from 'dexie';

export const db = new Dexie('FoldSpaceDB');

db.version(1).stores({
  // 主表：时空薄片
  folds: '++id, timestamp, emotion, tags, collapsed',
  // 废纸篓（软删除保留30天）
  trash: '++id, deletedAt, expiresAt',
  // 用户配置与文件句柄
  settings: 'key'
});

// 废纸篓自动清理（启动时执行）
export async function cleanTrash() {
  const now = Date.now();
  await db.trash.where('expiresAt').below(now).delete();
}
```

### 2.2 文件系统 API

```javascript
// fileSystem.js

// 请求用户选择保存目录（首次使用）
export async function requestSaveDirectory() {
  try {
    const dirHandle = await window.showDirectoryPicker({
      mode: 'readwrite',
      startIn: 'documents'
    });

    // 验证权限
    if ((await dirHandle.queryPermission({ mode: 'readwrite' })) === 'granted') {
      await db.settings.put({ key: 'saveDir', value: dirHandle });
      return dirHandle;
    }
  } catch (err) {
    console.log('用户拒绝或浏览器不支持，降级为下载模式');
    return null;
  }
}

// 保存数据到本地文件
export async function saveToFile(dirHandle = null) {
  const data = {
    format: 'fold-space-v1',
    exportedAt: new Date().toISOString(),
    version: '1.0.0',
    folds: await db.folds.toArray(),
    trash: await db.trash.toArray(),
    settings: {
      lastView: await db.settings.get('lastView'),
      cameraPos: await db.settings.get('cameraPos')
    }
  };

  const blob = new Blob([JSON.stringify(data, null, 2)], {
    type: 'application/fold-space'
  });

  // 现代浏览器：直接写入文件
  if (dirHandle || (dirHandle = await db.settings.get('saveDir')?.value)) {
    try {
      const fileHandle = await dirHandle.getFileHandle('my-universe.fold', { create: true });
      const writable = await fileHandle.createWritable();
      await writable.write(blob);
      await writable.close();
      return { success: true, mode: 'filesystem' };
    } catch (e) {
      console.error('文件写入失败，降级为下载', e);
    }
  }

  // 降级方案：传统下载
  const url = URL.createObjectURL(blob);
  const a = document.createElement('a');
  a.href = url;
  a.download = `fold-space-${new Date().toLocaleDateString()}.fold`;
  document.body.appendChild(a);
  a.click();
  document.body.removeChild(a);
  URL.revokeObjectURL(url);

  return { success: true, mode: 'download' };
}
```

## 三、核心数据流

### 3.1 日常操作流程

```text
用户手势操作（新建/编辑/揉碎）
    ↓
内存中的 Three.js 对象（实时渲染）
    ↓
Dexie.js 事务写入 IndexedDB（< 50ms）
    ↓
防抖自动保存（debounce 5秒）→ 写入本地文件
```

### 3.2 启动加载流程

```javascript
// App 启动时
export async function initStorage() {
  // 1. 清理过期废纸篓
  await cleanTrash();

  // 2. 检查 IndexedDB 是否有数据
  const foldCount = await db.folds.count();

  if (foldCount > 0) {
    // 本地有数据，直接加载到 Three.js 场景
    return { source: 'indexedDB', count: foldCount };
  }

  // 3. 本地无数据，尝试从文件恢复
  try {
    const [fileHandle] = await window.showOpenFilePicker({
      types: [{ accept: { 'application/fold-space': ['.fold'] } }],
      multiple: false
    });

    const file = await fileHandle.getFile();
    const data = JSON.parse(await file.text());

    // 恢复数据
    await db.folds.bulkAdd(data.folds || []);
    await db.trash.bulkAdd(data.trash || []);

    // 保存文件句柄以便自动保存
    await db.settings.put({ key: 'saveDir', value: await fileHandle.getDirectory() });

    return { source: 'file', count: data.folds?.length || 0 };
  } catch {
    // 新用户，创建欢迎数据
    await createWelcomeFold();
    return { source: 'new', count: 1 };
  }
}
```

## 五、自动保存策略

```javascript
// autoSave.js
import { debounce } from 'lodash-es'; // 或自定义防抖

// 防抖保存（用户停止操作5秒后触发）
const debouncedFileSave = debounce(async () => {
  const dirHandle = await db.settings.get('saveDir')?.value;
  if (dirHandle) {
    await saveToFile(dirHandle);
    showToast('已自动保存到本地文件');
  }
}, 5000);

// 每次数据变更后调用
export function triggerAutoSave() {
  // 立即保存到 IndexedDB（快速）
  // IndexedDB 写入是异步的，Dexie 自动处理

  // 延迟保存到文件（慢速但持久）
  debouncedFileSave();
}

// 紧急保存（关闭页面前）
window.addEventListener('beforeunload', async (e) => {
  debouncedFileSave.flush(); // 立即执行防抖队列
});
```

## 六、废纸篓机制（软删除）

```javascript
// trash.js

// 坍缩操作（揉碎）
export async function collapseFold(foldId) {
  const fold = await db.folds.get(foldId);
  if (!fold) return;

  // 1. 加入废纸篓（保留30天）
  await db.trash.add({
    ...fold,
    originalId: foldId,
    deletedAt: Date.now(),
    expiresAt: Date.now() + (30 * 24 * 60 * 60 * 1000) // 30天后自动物理删除
  });

  // 2. 从主表删除
  await db.folds.delete(foldId);

  // 3. 触发保存
  triggerAutoSave();

  return { success: true, message: '已坍缩成奇点，30天内可恢复' };
}

// 白洞恢复（从废纸篓还原）
export async function restoreFromTrash(trashId) {
  const item = await db.trash.get(trashId);
  if (!item) throw new Error('奇点已蒸发');

  const { originalId, deletedAt, expiresAt, ...foldData } = item;

  // 恢复时可能ID冲突，使用原ID或新ID
  await db.folds.add({ ...foldData, id: originalId });
  await db.trash.delete(trashId);

  triggerAutoSave();
  return { success: true };
}

// 彻底粉碎（不可逆删除）
export async function pulverizeTrash(trashId) {
  await db.trash.delete(trashId);
  triggerAutoSave();
}
```

## 七、隐私与加密（可选）

```javascript
// crypto.js
// 使用 Web Crypto API，纯前端 AES-GCM

export async function encryptFoldData(data, password) {
  const encoder = new TextEncoder();
  const salt = window.crypto.getRandomValues(new Uint8Array(16));

  // 从密码派生密钥
  const keyMaterial = await window.crypto.subtle.importKey(
    'raw',
    encoder.encode(password),
    'PBKDF2',
    false,
    ['deriveBits', 'deriveKey']
  );

  const key = await window.crypto.subtle.deriveKey(
    { name: 'PBKDF2', salt, iterations: 100000, hash: 'SHA-256' },
    keyMaterial,
    { name: 'AES-GCM', length: 256 },
    false,
    ['encrypt']
  );

  const iv = window.crypto.getRandomValues(new Uint8Array(12));
  const encrypted = await window.crypto.subtle.encrypt(
    { name: 'AES-GCM', iv },
    key,
    encoder.encode(JSON.stringify(data))
  );

  return {
    salt: Array.from(salt),
    iv: Array.from(iv),
    data: Array.from(new Uint8Array(encrypted)),
    algorithm: 'AES-GCM-256'
  };
}

// 保存时包裹
export async function saveEncrypted(password) {
  const rawData = {

}
```

## 八、兼容性处理

```javascript
// 特性检测与降级
export function checkStorageSupport() {
  return {
    indexedDB: 'indexedDB' in window,
    fileSystem: 'showDirectoryPicker' in window,
    filePick: 'showOpenFilePicker' in window,
    crypto: 'crypto' in window && 'subtle' in window.crypto,
    vibration: 'vibrate' in navigator
  };
}

// 根据支持度选择模式
export async function initWithFallback() {
  const support = checkStorageSupport();

  if (!support.indexedDB) {
    alert('您的浏览器不支持 IndexedDB，无法使用褶宇宙');
    return false;
  }

  if (!support.fileSystem) {
    console.log('降级模式：将使用传统文件下载/上传');
    // 显示手动导出/导入按钮
  }

  return true;
}
```

## 十、快速接入代码

```javascript
// 在 React/Vue 入口文件中
import { db, initStorage, triggerAutoSave } from './db';
import { requestSaveDirectory } from './fileSystem';

// 应用启动
onMounted(async () => {
  const status = await initStorage();

  // 如果是新用户，引导选择保存位置
  if (status.source === 'new') {
    const hasDir = await requestSaveDirectory();
    if (!hasDir) {
      alert('您选择了手动模式，请记得定期点击"导出"备份数据');
    }
  }

  // 监听所有数据变更（示例）
  db.folds.hook('creating', triggerAutoSave);
  db.folds.hook('updating', triggerAutoSave);
  db.folds.hook('deleting', triggerAutoSave);
});
```
